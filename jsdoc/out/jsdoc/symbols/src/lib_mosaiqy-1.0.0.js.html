 34      * @function
 35      * @param { String } ua     Current user agent specific string
 36      * @param { String } prop   The property we want to check
 37      * 
 38      * @returns { Object }
 39      * <pre>
 40      *      isEnabled       : True if acceleration is available, false otherwise;
 41      *      transitionEnd   : Event available on current browser;
 42      *      duration        : Vendor specific CSS property.
 43      * </pre>
 44      * 
 45      * @description
 46      * Detect if GPU acceleration is enabled for transitions.
 47      * code gist mantained at https://gist.github.com/892739

 49     GPUAcceleration = (function(ua, prop) {
 50     
 51         var div     = document.createElement('div'),
 52             cssProp = function(p) {
 53                 return p.replace(/([A-Z])/g, function(match, upper) {
 54                     return "-" + upper.toLowerCase();
 55                 });
 56             },
 57             vendorProp,
 58             uaList  = {
 59                 msie    : 'MsTransition',
 60                 opera   : 'OTransition',
 61                 mozilla : 'MozTransition',
 62                 webkit  : 'WebkitTransition'
 63             };
 64             
 65         for (var b in uaList) {
 66             if (uaList.hasOwnProperty(b)) {
 67                 if (ua[b]) { vendorProp = uaList[b]; }
 68             }
 69         }
 70                 
 71         return {
 72             isEnabled       : (function(s) {
 73                 return !!(s[prop] || vendorProp in s  || (ua.opera && parseFloat(ua.version) > 10.49));
 74             }(div.style)),
 75             transitionEnd   : (function() {
 76                 return (ua.opera)
 77                     ? 'oTransitionEnd'

 85     /**
 86      * @function
 87      * @description
 88      * This algorithm is described in http://en.wikipedia.org/wiki/Knuth_shuffle
 89      * The main purpose is to ensure an equally-distributed animation sequence, so
 90      * every entry point can have the same probability to be chosen without
 91      * duplicates.
 92      *
 93      * @returns { Array } A shuffled array of entry points.
 94      */
 95     shuffledFisherYates = function(len) {
 96         var i, j, tmp_i, tmp_j, shuffled = [];
 97         
 98         i = len;
 99         for (j = 0; j < i; j++) { shuffled[j] = j; }
100         while (--i) {
101             /*
102              * [~~] is the bitwise op quickest equivalent to Math.floor()
103              * http://jsperf.com/bitwise-not-not-vs-math-floor
104              */
105             j           = ~~(Math.random() * (i+1));
106             tmp_i       = shuffled[i];
107             tmp_j       = shuffled[j];
108             shuffled[i] = tmp_j;
109             shuffled[j] = tmp_i;
110         }
111             
112         return shuffled;
113     },

151         /**
152          * @private
153          * @name Mosaiqy#_createTemplate
154          * @param { Number } index The index of JSON data array
155          * @description
156          * 
157          * The method merges the user-defined template with JSON data associated
158          * for a given index and it's called both at initialization and at every
159          * animation loop.
160          * 
161          * @returns { jQuery } HTML Nodes to inject into the document
162          */
163         _createTemplate = function(index) {
164             var tpl = '';
165             if (typeof _tplCache[index] === 'undefined') {
166                 _tplCache[index] = _s.template.replace(/\$\{([^\}]+)\}/gm, function(data, key) {
167                     
168                     /**
169                      * if key has one or more dot then a nested key has been requested
170                      * and a while loop goes in-depth over the JSON
171                      */
172                     var value = (function() {
173                         var par = key.split('.'), len = 0, val;
174                         if (par.length) {
175                             val = _s.data[index];
176                             par = par.reverse();
177                             len = par.length;
178                                 
179                             while (len--) {  val = val[par[len]] || { }; }
180                             return (typeof val === "string")? val : key;
181                         }
182                         return _s.data[index][key];
183                     }());
184                     
185                     if (typeof value === 'undefined') {
186                         return key;
187                     }
188                     return value;
189                 });
190             }
191             
192             tpl = _tplCache[index];
193             if (typeof window.innerShiv === 'function') {
194                 tpl = window.innerShiv(tpl);
195             }
196             
197             return $(tpl);

238         /**
239          * @private
240          * @name Mosaiqy#_getPoints
241          * @description
242          * 
243          * _getPoints object stores 4 information
244          * <ol>
245          *   <li>The direction of movement (css property)</li>
246          *   <li>The selection of nodes to move (i.e in a 3x4 grid, point 4 and 5 have
247          *      to move images 2,6,10)</li>
248          *   <li>The position in which we want to inject-before the new element (except
249          *      for the last element which needs to be injected after)</li>
250          *   <li>The position (top and left properties) of entry tile</li>
251          * </ol>
252          * 
253          * <code><pre>
254          *    [0,8,1,9,2,10,3,11, 0,4,4,8,8,12*]    * = append after
255          *    
256          *        0   2   4   6
257          *    8 |_0_|_1_|_2_|_3_| 9
258          *   10 |_4_|_5_|_6_|_7_| 11
259          *   12 |_8_|_9_|_10|_11| 13    
260          *        1   3   5   7
261          * </pre></code>
262          * 
263          * <p>
264          * In earlier versions of this algorithm, the order of nodes was counterclockwise
265          * (tlbr) and then alternating (tblr). Now this enumeration pattern (alternating
266          * tb and lr) performs a couple of improvements on code logic and on readability:
267          * </p>
268          *
269          * <ol>
270          *   <li>Given an even-indexed node, the next adjacent index has the same selector:<br />
271          *      e.g. points[8] = li:nth-child(n+1):nth-child(-n+4) [0123]<br />
272          *           points[9] = li:nth-child(n+1):nth-child(-n+4) [0123]<br />
273          *      (it's easier to retrieve this information)</li>
274          *   <li>If a random point is even (i & 1 === 0) then the 'direction' property of node
275          *      selection is going to be increased during slide animation. Otherwise is going
276          *      to be decreased and then we remove first or last element (if random number is 9,
277          *      then the collection has to be [3210] and not [0123], since we always remove the
278          *      disappeared node when the animation has completed.)</li>
279          * </ol>
280          *
281          * @example
282          *    Another Example (4x2)
283          *    [0,6,1,7, 0,2,2,4,4,6,6,8*]   * = append after
284          *
285          *        0   2
286          *    4 |_0_|_1_| 5
287          *    6 |_2_|_3_| 7
288          *    8 |_4_|_5_| 9
289          *   10 |_6_|_7_| 11
290          *        1   3
291          */
292         _getPoints = function() {
293             
294             var c, n, s, /* internal counters */
295                 selectors = {
296                     col : "li:nth-child($0n+$1)",
297                     row : "li:nth-child(n+$0):nth-child(-n+$1)"
298                 };
299             
300             /* cols information */
301             for (n = 0; n < _s.cols; n = n + 1) {
302                 
303                 s = selectors.col.replace(/\$(\d)/g, function(selector, i) {
304                     return [_s.cols, n + 1][i]; });
305                     
306                 _points.push({ prop: 'top', selector : s, node : n,
307                     position : {
308                         top     : -(_thumbSize.h),
309                         left    : _thumbSize.w * n
310                     }
311                 });
312                 _points.push({ prop: 'top', selector : s, node : _s.cols * (_s.rows - 1) + n,
313                     position : {
314                         top     : _thumbSize.h * _s.rows,
315                         left    : _thumbSize.w * n
316                     }
317                 });
318             }
319                 
320             /* rows information */
321             for (c = 0, n = 0; n < _s.rows; n = n + 1) {
322                 
323                 s = selectors.row.replace(/\$(\d)/g, function(selector, i) {
324                     return [c + 1, c + _s.cols][i]; });
325                     
326                 _points.push({ prop: 'left', selector : s, node : c,
327                     position : {
328                         top     : _thumbSize.h * n,
329                         left    : -(_thumbSize.w)
330                     }
331                 });
332                 _points.push({ prop: 'left', selector : s, node : c += _s.cols,
333                     position : {
334                         top     : _thumbSize.h * n,
335                         left    : _thumbSize.w * _s.cols
336                     }
337                 });
338             }
339                 
340             _points[_points.length - 1].node -= 1;
341             appDebug("groupCollapsed", 'points information');
342             appDebug(($.browser.mozilla)?"table":"dir", _points);
343             appDebug("groupEnd");
344         },

349          * @private
350          * @name Mosaiqy#_animateSelection
351          * @return a deferred promise
352          * @description
353          *
354          * This method runs the animation.
355          */
356         _animateSelection = function() {
357             
358             var rnd, tpl, referral, node, animatedSelection, isEven,
359                 dfd = $.Deferred();
360                 
361             appDebug("groupCollapsed", 'call animate()');
362             appDebug("info", 'Dataindex is', _s.indexData);
363                 
364                 
365             /**
366              * Get the entry point from shuffled array
367              */ 
368             rnd = _entryPoints.pop();
369             isEven = ((rnd & 1) === 0);
370             
371             animatedSelection = _cnt.find(_points[rnd].selector);
372             /**
373              * append new li element
374              * if the random entry point is the last one then we append the
375              * new node after the last li, otherwise we place it before.
376              */
377             referral    = _li.eq(_points[rnd].node);
378             node        = (rnd < _points.length - 1)?
379                   $('<li />').insertBefore(referral)
380                 : $('<li />').insertAfter(referral);
381                 
382             node.data('mosaiqy-index', _s.indexData);
383                 
384                 
385             /**
386              * Generate template to append with user data
387              */
388             tpl = _createTemplate(_s.indexData);
389             tpl.appendTo(node.css(_points[rnd].position));
390                 
391             appDebug("info", "Random position is %d and its referral is node", rnd, referral);
392                 
393             /**
394              * Looking for images inside template fragment, wait the deferred
395              * execution and checking a promise status.
396              */
397             $.when(node.find('img').mosaiqyImagesLoad(_s.loadTimeout))
398             /**
399              * No image/s can be loaded, remove the node inserted, then call
400              * again the _animate method
401              */
402             .fail(function() {
403                 appDebug("warn", 'Skip dataindex %d, call _animate()', _s.indexData);
404                 appDebug("groupEnd");
405                 node.remove();
406                 dfd.reject();
407             })
408             /**
409              * Image/s inside template fragment have been successfully loaded so
410              * we can apply the slide transition on the selected nodes and the
411              * added node
412              */ 
413             .done(function() {
414                 var prop            = _points[rnd].prop,
415                     amount          = (prop === 'left')? _thumbSize.w : _thumbSize.h,
416                     /**
417                      * @ignore
418                      * add new node into animatedNodes collection and change
419                      * previous collection
420                      */
421                     animatedNodes   = animatedSelection.add(node),
422                     animatedQueue   = animatedNodes.length,
423                     move = {};
424                 
425                 move[prop] = '+=' + (isEven? amount : -amount) + 'px';
426                 appDebug("log", 'Animated Nodes:', animatedNodes);
427                 
428                 /**
429                  * $.animate() function has been extended to support css transition
430                  * on modern browser. For this reason I cannot use deferred animation,
431                  * because if GPUacceleration is enabled the code will not use native
432                  * animation.
433                  *
434                  * See code below
435                  */
436                 animatedNodes.animate(move , _s.animationSpeed,
437                     function() {
438                         var len;
439                             
440                         if (--animatedQueue) { return; }
441                             
442                         /**
443                          * Opposite node removal. "Opposite" is related on sliding direction
444                          * e.g. on 2->[159] (down) opposite has index 9
445                          *      on 3->[159] (up) opposite has index 1
446                          */
447                         if (isEven) {
448                             animatedSelection.last().remove();
449                         }
450                         else {
451                             animatedSelection.first().remove();
452                         }
453                             
454                         appDebug("log", 'Animated Selection:', animatedSelection);
455                         animatedSelection = (isEven) 
456                             ? animatedSelection.slice(0, animatedSelection.length - 1)
457                             : animatedSelection.slice(1, animatedSelection.length);
458                             
459                         appDebug("log", 'Animated Selection:', animatedSelection);
460                             
461                         /**
462                          * <p>Node rearrangement when animation affects a column. In this case
463                          * a shift must change order inside li collection, otherwise the 
464                          * subsequent node selection won't be properly calculated.
465                          * Algorithm is quite simple:</p>
466                          *
467                          * <ol>
468                          *   <li>The offset displacement of shifted nodes is always
469                          *       determined by the number of columns except when shift direction is
470                          *       bottom-up: in fact the last node of animatedSelection collection
471                          *       represents an exception because its position is affected by the
472                          *       presence of the new node (placed just before it);</li>
473                          *   <li>offset is negative on odd entry point (down and right) and
474                          *       positive otherwise (top and left);</li>
475                          *   <li>at each iteration we retrieve the current li nodes in the
476                          *       grid so we can work with actual node position.</li>
477                          * </ol>
478                          * 
479                          * <p>If the animation affected a row, rearrangement of nodes is not needed
480                          * at all because insertion is sequential, thus the new node and shifted
481                          * nodes already have the right index.</p>
482                          */
483                         if (prop === 'top') {
484                             len = animatedSelection.length;
485                             
486                             animatedSelection.each(function(i) {
487                                 var node, curpos, offset, newpos;
488                                 
489                                 /**
490                                  * @ignore
491                                  * Retrieve node after each new insertion and rearrangement
492                                  * of selected animating nodes 
493                                  */ 
494                                 _li     = _cnt.find("li:not(.mosaiqy-zoom)");
495                                 
496                                 node    = $(this);
497                                 curpos  = _li.index(node);
498                                 offset  = (isEven) ? _s.cols : -(_s.cols - ((1 === len - i)? 0 : 1));
499                                         
500                                 if (!!offset) { 
501                                     newpos  = curpos + offset;
502                                     if (newpos < _li.length) {
503                                         node.insertBefore(_li.eq(newpos));
504                                     }
505                                     else {
506                                         node.appendTo(_ul);
507                                     }
508                                 }
509                             
510                             });
511                         }
512                         appDebug("groupEnd");
513                         dfd.resolve();
514                     }
515                 );
516             });
517             
518             return dfd.promise();
519         },

523         /**
524          * @private
525          * @name Mosaiqy#_animationCycle
526          * @description
527          * 
528          * <p>The method runs the animation and check some private variables to
529          * allow cycle and animation execution. Every time the animation has
530          * completed successfully, the JSON index and node collection are updated.</p>
531          * 
532          * <p>Animation interval is not executed on mouse enter (_animationPaused)
533          * or when animation is still running.</p>
534          */ 
535         _animationCycle = function() {
536             if (!_animationPaused && !_animationRunning) {
537                 
538                 _animationRunning = true;
539                 
540                 if (_entryPoints.length === 0) {
541                     _entryPoints = shuffledFisherYates(_points.length);
542                     appDebug("info", 'New entry point shuffled array', _entryPoints);
543                 }
544                 
545                 appDebug("info", 'Animate selection');
546                 _incrementIndexData();
547                 
548                 $.when(_animateSelection())
549                     /**
550                      * In all cases dataIndex is increased and the animationRunning
551                      * state is set to false so animation could continue.
552                      */
553                     .then(function() {
554                         _s.indexData = _s.indexData + 1;
555                         _animationRunning = false;
556                         appDebug("info", 'End animate selection');
557                     })
558                     /**
559                      * If a thumbnail was not loaded within the defined limit then animation
560                      * should not wait another delay. We call soon the method again.
561                      */
562                     .fail(function() {
563                         _animationCycle();                        
564                     })
565                     /**
566                      * Thumbnail was loaded. Update the reference of list-items (changed)
567                      * on stage and call the method again after timeout.
568                      */
569                     .done(function() {
570                         _li = _ul.find('li:not(.mosaiqy-zoom)');
571                         _intvAnimation = setTimeout(function() {
572                             _animationCycle();
573                         }, _s.animationDelay);
574                     });
575             }
576             else {
577                 _intvAnimation = setTimeout(function() {
578                     _animationCycle();
579                 }, _s.animationDelay * 2);
580             }
581         },

612         /**
613          * @private
614          * @name Mosaiqy#_incrementIndexData
615          * @description
616          * 
617          * <p>The main purpose is to correctly increment the indexData for the JSON
618          * data retrieval. If user choosed "avoidDuplicate" option, then the method
619          * checks if a requested image is already on stage. If so, a loop starts
620          * looking for the first image not in stage, increasing the dataIndex.</p>
621          */
622         _incrementIndexData     = function() {
623             
624             var safe    = _s.data.length,
625                 stage   = [];
626                 
627             if (_s.indexData === _s.data.length) {
628                 if (!_s.loop) {
629                     return _pauseAnimation();
630                 }
631                 else {
632                     _s.indexData = 0;
633                 }
634             }
635                 
636             if (_s.avoidDuplicates) {
637                 appDebug('info', "Avoid Duplicates");
638                 _li.each(function() {
639                     var i = $(this).data('mosaiqy-index');
640                     stage[i] = i;
641                 });
642                 appDebug('info', "Now on stage: ", stage);
643                 
644                 while (safe--) {
645                     if (typeof stage[_s.indexData] !== 'undefined') {
646                         appDebug('info', "%d already exist (skip)", _s.indexData)
647                         _s.indexData = _s.indexData + 1;
648                         if (_s.indexData === _s.data.length) {
649                             if (!_s.loop) {
650                                 return _pauseAnimation();
651                             }
652                             else {
653                                 _s.indexData = 0;
654                             }
655                         }
656                         continue;
657                     }
658                     appDebug('info', "%d not in stage (ok)", _s.indexData);
659                     break;
660                 }
661             }

666         /**
667          * @private
668          * @name Mosaiqy#_setNodeZoomEvent
669          * @description
670          * 
671          * <p>This method manages the zoom main events by some scoped internal functions.</p>
672          * 
673          * <p><code>closeZoom</code> is called when user clicks on "Close" button over a zoom
674          * image or when another thumbanail is choosed and another zoom is currently opened.
675          * The function stops all running transitions (if any) and it closes the zoom container
676          * while changing opacity of some elements (close button, image caption). At the end of
677          * animation it removes some internal classes and the li node that contained the zoom.</p>
678          *
679          * <p>The function <code>closeZoom</code> returns a deferred promise object, so it can be
680          * called in a synchronous code queue inside other functions, ensuring all operation have
681          * been successfully completed.</p>
682          *
683          * <p><code>viewZoom</code> is called when the previous function <code>createZoom</code>
684          * successfully created the zoom container into the DOM. The function creates the zoom image
685          * and the closing button binding the click event. If image is not in cache the zoom is opened
686          * with a slideDown effect with a waiting loader.</p>
687          * 
688          * <p><code>createZoom</code> calls the <code>closeZoom</code> function (if any zoom images
689          * are currently opened) then creates the zoom container into the DOM and then scroll the page
690          * until the upper bound of the thumbnail choosed has reached (unless scrollzoom option is set to
691          * false). When scrolling effect has completed then <code>viewZoom</code> function is called.</p>
692          */ 
693         _setNodeZoomEvent   = function(node) {
694                 
695             var nodezoom, $this, i, zoomRunning,
696                 zoomFigure, zoomCaption, zoomCloseBtt,
697                 pagePos, thumbPos, diffPos;
698             
699             function closeZoom() {
700                 var dfd = $.Deferred();
701                 
702                 if ((nodezoom || { }).length) {
703                     appDebug("log", 'closing previous zoom');

948                 imgToComplete = (_s.cols * _s.rows) - _li.length;

1038             imgLength   = this.length,
1039             loaded      = [],
1040             failed      = [],
1041             timeout     = to || 8419.78;
1042             /* waiting about 8 seconds before discarding image */
1043             
1044         appDebug("groupCollapsed", 'Start deferred load of %d image/s:', imgLength);

1060                          * the image has loaded or if error event has been triggered.

1062                         imageDfd    = $.Deferred(),

1066                         $(i).one('load.mosaiqy', function() {
1067                                 clearInterval(intv);

1070                             .bind('error.mosaiqy', function() {
1071                                 clearInterval(intv);

1084                 })
1085                 .fail(function() {
1086                     failed.push(i.src);
1087                     appDebug("warn", 'Not loaded', i.src);
1088                 })
1089                 .always(function() {
1090                     imgLength = imgLength - 1;
1091                     if (imgLength === 0) {
1092                         appDebug("groupEnd");
1093                         if (failed.length) {
1094                             dfd.reject();
1095                         }
1096                         else {
1097                             dfd.resolve();
